function drawPoint(a, b, c) {
    let v = 0, h = 0, x = 0, y = 0;
    for (let t = 0; t < 12.56; t += .05) {
        x = a * Math.cos(t) + b * Math.cos(c * t);
        y = a * Math.sin(t) - b * Math.sin(c * t);
        v = (dx - y * 2) * ez;
        h = (dx + x * 2) * ez;
        myContext.fillRect(h, v, 2, 2);
    }
}

// graphic functions
function F1(x, y) {
    return (Math.sin(x) + Math.cos(y)) / 60;
}

function F2(x, y) {
    return Math.sin(x + y) / 25;
}

function F3(x, y) {
    var c = 50 - (x * x) - (y * y);
    return Math.sqrt(c * (_sgn(c) + 1)) / 30;
}

function F4(x, y) {
    var c = (x * x) + (y * y) + .00001;
    var r = ((_sgn(parseInt(13.0 / c))) / 3.0) + ((_sgn(parseInt(35.0 / c))) / 15.0);
    return r;
}

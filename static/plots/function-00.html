Simple Canvas example


0 SCREEN 7 : GOTO 10
1 R=SQR(X*X+Y*Y)*1.5 : IF R=0 THEN F=90: GOTO 3
2 F=ABS(90*SIN(R)/R)
3 A=INT(10*X+160-5*Y)
4 B=INT(5*Y+2.5*X+93)
5 RETURN
10 FOR Y=-10 TO 10 STEP 0.2
30 FOR X=-10 TO 10 STEP 0.3
40 GOSUB 1
46 PSET(A,B-F), 15
50 NEXT X
60 NEXT Y
